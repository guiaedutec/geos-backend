def delete_school
  user = current_user
  if user.admin?
    school = (!params['id'].nil?) ? School.find_by(id: params['id']) : nil
    if (!school.nil?)
        school.delete
        render json: {status: 'SUCCESS', message:'School deleted', data:school},status: :ok
    else
        render json: {status: 'ERROR', message:'School not found', data: nil},status: :not_found
    end  
  end #if-user.admin?
end

def edit_school
  user = current_user
  if user.admin?
    if !(params[:name].nil? || params[:country_name].nil? || params[:province_name].nil? || params[:state_name].nil? || params[:affiliation_name].nil?)
      # Verificar se a instituição já está no banco
      school = School.where(id: params[:id]).first
      if (!school.nil?)
          school.update_attributes(
            name: params[:name],
            affiliation_id: BSON::ObjectId.from_string(params[:affiliation_id]),
            affiliation_name: params[:affiliation_name],
            country_id: BSON::ObjectId.from_string(params[:country_id]),
            country_name: params[:country_name],
            province_id: BSON::ObjectId.from_string(params[:province_id]),
            province_name: params[:province_name],
            state_id: BSON::ObjectId.from_string(params[:state_id]),
            state_name: params[:state_name],
            city_id: !(params[:city_id].nil?) ? BSON::ObjectId.from_string(params[:city_id]) : nil,
            city_name: !(params[:city_name].nil?) ? params[:city_name] : nil)
          render json: {status: 'SUCCESS', message:'School updated', data:school},status: :ok
      else
          render json: {status: 'ERROR', message:'School not found'},status: :not_found
      end #if (!school.nil?  
    end #if !(params[:name].nil?
  end #if-user.admin?
end

def save_school
  user = current_user
  params.permit!
  if user.admin?
    if !(params[:name].nil? || params[:country_name].nil? || params[:province_name].nil? || params[:state_name].nil? || params[:affiliation_name].nil?)
      # Verificar se a instituição já está no banco
      school = School.where(
        name: params[:name],
        affiliation_name: params[:affiliation_name],
        country_name: params[:country_name],
        province_name: params[:province_name],
        state_name: params[:state_name],
        city_name: !(params[:city_name].nil?) ? params[:city_name] : nil).first
      if (school.nil?) # Insert
        school = School.new(
                        name: params[:name],
                        affiliation_id: BSON::ObjectId.from_string(params[:affiliation_id]),
                        affiliation_name: params[:affiliation_name],
                        country_id: BSON::ObjectId.from_string(params[:country_id]),
                        country_name: params[:country_name],
                        province_id: BSON::ObjectId.from_string(params[:province_id]),
                        province_name: params[:province_name],
                        state_id: BSON::ObjectId.from_string(params[:state_id]),
                        state_name: params[:state_name],
                        city_id: !(params[:city_id].nil?) ? BSON::ObjectId.from_string(params[:city_id]) : nil,
                        city_name: !(params[:city_name].nil?) ? params[:city_name] : nil)
        if school.save
          render json: {status: 'SUCCESS', message:'Saved school', data:school},status: :ok
        else
          render json: {status: 'ERROR', message:'School not saved', data: nil},status: :unprocessable_entity
        end
      else
        render json: {status: 'ERROR', message:'School already exists', data: school},status: :unprocessable_entity
      end
    end #if (params
  end #if-user.admin?
end
